#
# Función compuesta de varias subfunciones
# Principio de responsabilidad simple:
# cada función hace una tarea y solo una tarea
#
def complex_function(x):
    def constant():
        return 1

    def first():
        return x

    def second():
        return x ** 2

    result = constant() + first() + second()
    return result


complex_function(2)
