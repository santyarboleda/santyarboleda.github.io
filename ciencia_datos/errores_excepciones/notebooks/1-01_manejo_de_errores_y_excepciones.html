def sqrt(x):
    if x < 0:
        raise ValueError("x must be > 0")
    try:
        return x ** 0.5

    except TypeError:
        print("x must be a float or integer")


display(
    sqrt(1.0),
    sqrt(-1.0),
    sqrt("1"),
)
